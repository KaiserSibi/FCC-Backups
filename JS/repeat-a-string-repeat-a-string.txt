function repeatStringNumTimes(str, num) {
  if (num <= 0){
    return str = "";
  }
  else{
    let tempStr = ""
    for (let i = 1; i <= num; i++){
      tempStr += str;
    }
    str = tempStr;
    return str;
  }
}
repeatStringNumTimes("abc", 2);
